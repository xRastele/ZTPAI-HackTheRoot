CREATE TABLE users (
    id_users SERIAL PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
	is_admin BOOLEAN DEFAULT FALSE;
);

CREATE TABLE notifications (
    id_notifications SERIAL PRIMARY KEY,
    id_users INTEGER REFERENCES users(id_users),
    notifications_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    notifications_text TEXT
);

CREATE TABLE leaderboard (
    id_users INTEGER PRIMARY KEY REFERENCES users(id_users),
    points INTEGER NOT NULL DEFAULT 0
);

CREATE TABLE progress (
    id_progress SERIAL PRIMARY KEY,
    id_users INTEGER REFERENCES users(id_users),
    id_challenges INTEGER REFERENCES challenges(id_challenges),
    is_completed BOOLEAN,
    completed_date TIMESTAMP
);

CREATE TABLE challenges (
    id_challenges SERIAL PRIMARY KEY,
    id_rewards INTEGER REFERENCES rewards(id_rewards)
);

CREATE TABLE rewards (
    id_rewards SERIAL PRIMARY KEY,
    points INTEGER NOT NULL
);

CREATE TABLE learning_paths (
    id_learning_paths SERIAL PRIMARY KEY,
    description TEXT
);

CREATE TABLE learning_paths_challenges (
    id_learning_paths_challenges SERIAL PRIMARY KEY,
    id_learning_paths INTEGER REFERENCES learning_paths(id_learning_paths),
    id_challenges INTEGER REFERENCES challenges(id_challenges)
);

CREATE TABLE settings (
    id_users INTEGER PRIMARY KEY REFERENCES users(id_users),
    referral_code VARCHAR(50),
    referral_count INTEGER DEFAULT 0
);
